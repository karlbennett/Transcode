00345     uint8_t buffer[BUFFER_SIZE];

00347     memset(buffer, 0, BUFFER_SIZE);
00348     // Copy the packet data into the new buffer.
00349     memcpy(buffer, packet->data, packet->size);
00350     // Set the copy packets data and size.
00351     packetCopy.data = buffer;
00352     packetCopy.size = packet->size;
00353 
00354     // The frame pointer that will hold each new frame before it is placed in the vector.
00355     AVFrame *decodedFrame = NULL;
00356 
00357     // This will be set to 1 if a frame has successfully been decoded with the
00358     // avcodec_decode_audio4() function.
00359     int frameDecoded = 0;
00360 
00361     int bytesDecoded = 0; // The number of bytes that were decoded in each iteration.
00362 
00363     while (0 < packetCopy.size) {
00364         // Create a new frame to contain the decoded data if one is required.
00365         if (NULL == decodedFrame) decodedFrame = avcodec_alloc_frame();

00368         // Also record how many bytes were decoded because it might not have been all

00370         bytesDecoded = avcodec_decode_audio4(codecContext, decodedFrame,
00371                 &frameDecoded,
00372                 &packetCopy);
00373 
00374         // If there is an invalid data error throw a more specific exception.
00375         if (AVERROR_INVALIDDATA == bytesDecoded) {

00377             throw InvalidPacketDataException(errorMessage(bytesDecoded));
00378         }
00379 
00380         if (0 > bytesDecoded) {
00381 
00382             throw PacketDecodeException(errorMessage(bytesDecoded));
00383         }
00384 
00385         // If a frame was successfully decoded add it the vector to be returned and
00386         // set the pointer to null to indicate we need a new frame allocated.
00387         if (0 != frameDecoded) {
