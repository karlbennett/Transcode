00320 /**
00321  * Extract the media meta data of the given type from the libav AVFormatContext struct
00322  * using the provided get meta data function.
00323  *
00324  * @param videoFile - the av format context that contains the audio meta data to
00325  *          extract.
00326  * @param mediaType - the type of media stream that should have it's codec inspected.
00327  * @param metadataCallback - the function to use to get the correct meta data out of
00328  *          the codec.
00329  *
00330  * @return a vector containing all the extracted meta data.
00331  */
00332 template<typename T> std::vector<T> extractMetaData(
00333         const AVFormatContext *videoFile, AVMediaType mediatype,
00334         std::tr1::function<T(const AVStream&)> metadataCallback) {
00335 
00336     checkFormatContext(videoFile);
00337 
00338     std::vector<T> details;
00339 
00340     AVStream *stream;
00341     AVCodecContext *codecContext;
00342     AVMediaType codecType;
00343 
00344     // Iterate through the streams searching for any audio streams.
00345     for (int i = 0; i < videoFile->nb_streams; i++) {
00346 
00347         stream = videoFile->streams[i];
00348         if (NULL == stream) {
00349             throw FFMPEGException("A media stream was null.");
00350         }
00351 
00352         codecContext = videoFile->streams[i]->codec;
00353         if (NULL == codecContext) {
00354             throw FFMPEGException("A codec was null.");
00355         }
00356 
00357         codecType = codecContext->codec_type;
00358 
00359         // If we find a stream that matches the provided type use its details to
00360         // construct a struct of the requested type.
00361         if (codecType == mediatype) {
00362 
00363             details.push_back(metadataCallback(*stream));
00364         }
00365     }
00366 
00367     return details;
00368 }
